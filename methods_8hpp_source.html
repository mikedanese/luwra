17  /**
18  * Helper struct for wrapping user type methods
19  */
20  template <typename T>
21  struct MethodWrapper {
22  static_assert(
23  sizeof(T) == -1,
24  "Undefined template MethodWrapper"
25  );
26  };

28  // 'const volatile'-qualified methods
29  template <typename T, typename R, typename... A>
30  struct MethodWrapper<R (T::*)(A...) const volatile> {
31  using MethodPointerType = R (T::*)(A...) const volatile;
32  using FunctionSignature = R (const volatile T*, A...);
33 
34  template <MethodPointerType meth> static inline
35  R hook(const volatile T* parent, A&&... args) {
36  return (parent->*meth)(std::forward<A>(args)...);
37  }
38 
39  template <MethodPointerType meth> static inline
40  int invoke(State* state) {
41  return call<FunctionSignature>(state, hook<meth>);
42  }
43  };
