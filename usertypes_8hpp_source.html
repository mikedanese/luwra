109 /**
110  * User type T.
111  * Instances created using this specialization are allocated and constructed as full user data
112  * types in Lua. The default garbage-collecting hook will destruct the user type, once it has
113  * been marked.
114  */
115 template <typename U>
116 struct Value<U&> {
117  using T = internal::StripUserType<U>;
118 
119  static inline
120  U& read(State* state, int n) {
121  // T is unqualified, therefore conversion from T& to U& is allowed
122  return *internal::check_user_type<T>(state, n);
123  }
124 
125  template <typename... A> static inline
126  int push(State* state, A&&... args) {
127  void* mem = lua_newuserdata(state, sizeof(T));
128 
129  if (!mem) {
130  luaL_error(state, "Failed to allocate user type");
131  return -1;
132  }
133 
134  // Construct
135  new (mem) T {std::forward<A>(args)...};
136 
137  // Apply metatable for unqualified type T
138  internal::apply_user_type_meta_table<T>(state);
139 
140  return 1;
141  }
142 };

144 /**
145  * User type T.
146  * Instances created using this specialization are allocated as light user data in Lua.
147  * The default garbage-collector does not destruct light user data types.
148  */
149 template <typename U>
150 struct Value<U*> {
151  using T = internal::StripUserType<U>;
152 
153  static inline
154  U* read(State* state, int n) {
155  // T is unqualified, therefore conversion from T* to U* is allowed
156  return internal::check_user_type<T>(state, n);
157  }
158 
159  static inline
160  int push(State* state, T* instance) {
161  if (instance == nullptr)
162  return 0;
163 
164  // Push instance as light user data
165  lua_pushlightuserdata(state, instance);
166 
167  // Apply metatable for unqualified type T
168  internal::apply_user_type_meta_table<T>(state);
169 
170  return 1;
171  }
172 };

174 /**
175  * Register the metatable for user type `T`. This function allows you to register methods
176  * which are shared across all instances of this type.
177  *
178  * By default a garbage-collector hook and string representation function are added as meta methods.
179  * Both can be overwritten.
180  *
181  * \tparam U User type struct or class
182  *
183  * \param state Lua state
184  * \param methods Map of methods
185  * \param meta_methods Map of meta methods
186  */
187 template <typename U> static inline
188 void registerUserType(
189  State* state,
190  const std::map<const char*, CFunction>& methods = std::map<const char*, CFunction>(),
191  const std::map<const char*, CFunction>& meta_methods = std::map<const char*, CFunction>()
192 ) {
193  using T = internal::StripUserType<U>;
194 
195  // Setup an appropriate metatable name
196  internal::new_user_type_metatable<T>(state);
197 
198  // Register methods
199  if (methods.size() > 0 && meta_methods.count("__index") == 0) {
200  push(state, "__index");
201  lua_newtable(state);
202 
203  for (auto& method: methods) {
204  setFields(state, -1, method.first, method.second);
205  }
206 
207  lua_rawset(state, -3);
208  }
209 
210  // Register garbage-collection hook
211  if (meta_methods.count("__gc") == 0) {
212  setFields(state, -1, "__gc", &internal::destruct_user_type<T>);
213  }
214 
215  // Register string representation function
216  if (meta_methods.count("__tostring") == 0) {
217  setFields(state, -1, "__tostring", &internal::stringify_user_type<T>);
218  }
219 
220  // Insert meta methods
221  for (const auto& metamethod: meta_methods) {
222  setFields(state, -1, metamethod.first, metamethod.second);
223  }
224 
225  // Pop metatable off the stack
226  lua_pop(state, -1);
227 }

251  * is a signature `U(A...)` where `U` is the user type and `A...` its constructor parameters.
