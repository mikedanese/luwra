90 /**
91  * Define a template specialization of `Value` for `type` with a `retrf(State*, int)` which
92  * extracts it from the stack and a `pushf(State*, type)` which pushes the value on the stack again.
93  * This assumes that only one value will be pushed onto the stack.
94  */
95 #define LUWRA_DEF_VALUE(type, retrf, pushf) \
96  template <> \
97  struct Value<type> { \
98  static inline \
99  type read(State* state, int n) { \
100  return retrf(state, n); \
101  } \
102  \
103  static inline \
104  int push(State* state, type value) { \
105  pushf(state, value); \
106  return 1; \
107  } \
108  }

173  // Base for `Value<I>` specializations which uses `B` as transport unit, where `I` is smaller
174  // than `B`.
175  template <typename I, typename B>
176  struct NumericContainedValueBase {
177  static constexpr
178  bool qualifies =
179  // TODO: Remove warning about comparsion between signed and unsigned integers
180  std::numeric_limits<I>::max() <= std::numeric_limits<B>::max()
181  && std::numeric_limits<I>::lowest() >= std::numeric_limits<B>::lowest();
182 
183  static inline
184  I read(State* state, int index) {
185  return
186  static_cast<I>(NumericTransportValue<B>::read(state, index));
187  }
188 
189  static inline
190  int push(State* state, I value) {
191  NumericTransportValue<B>::push(state, static_cast<B>(value));
192  return 1;
193  }
194  };

196  // Base for `Value<I>` specializations which uses `B` as transport unit, where `I` is bigger
197  // than `B`.
198  template <typename I, typename B>
199  struct NumericTruncatingValueBase {
200  static inline
201  I read(State* state, int index) {
202  return static_cast<I>(NumericTransportValue<B>::read(state, index));
203  }
204 
205  static inline
206  int push(State*, I) {
207  static_assert(
208  sizeof(I) == -1,
209  "You must not use 'Value<I>::push' specializations which inherit from NumericTruncatingValueBase"
210  );
211 
212  return -1;
213  }
214  };

216  // Base for `Value<I>` specializations which uses `B` as transport unit
217  template <typename I, typename B>
218  using NumericValueBase =
219  typename std::conditional<
220  std::is_same<I, B>::value,
221  NumericTransportValue<B>,
222  typename std::conditional<
223  NumericContainedValueBase<I, B>::qualifies,
224  NumericContainedValueBase<I, B>,
225  NumericTruncatingValueBase<I, B>
226  >::type

283 /**
284  * An arbitrary value on an execution stack.
285  * Note: this value is only available as long as it exists on its originating stack.
286  */
287 struct Arbitrary {
288  State* state;
289  int index;
290 };
291 
292 template <>
293 struct Value<Arbitrary> {
294  static inline
295  Arbitrary read(State* state, int index) {
296  if (index < 0)
297  index = lua_gettop(state) + (index + 1);
298 
299  return Arbitrary {state, index};
300  }
301 
302  static inline
303  int push(State* state, const Arbitrary& value) {
304  lua_pushvalue(value.state, value.index);
305 
306  if (value.state != state)
307  lua_xmove(value.state, state, 1);
308 
309  return 1;
310  }
311 };

317  template <size_t I>
318  struct StackPusher<std::index_sequence<I>> {
319  template <typename... T> static inline
320  int push(State* state, const std::tuple<T...>& package) {
321  using R = typename std::tuple_element<I, std::tuple<T...>>::type;
322  return std::max(0, Value<R>::push(state, std::get<I>(package)));
323  }
324  };

326  template <size_t I, size_t... Is>
327  struct StackPusher<std::index_sequence<I, Is...>> {
328  template <typename... T> static inline
329  int push(State* state, const std::tuple<T...>& package) {
330  return
331  StackPusher<std::index_sequence<I>>::push(state, package)
332  + StackPusher<std::index_sequence<Is...>>::push(state, package);
333  }
