19  void push(State* state, int index, K&& key, V&& value, R&&... rest) {

25  template <typename K, typename V>
26  struct EntryPusher<K, V> {
27  static inline
28  void push(State* state, int index, K&& key, V&& value) {
29  assert(1 == luwra::push(state, key));
30  assert(1 == luwra::push(state, value));
31  lua_rawset(state, index < 0 ? index - 2 : index);
32  }
33  };

62  lua_getglobal(state, name.c_str());

64  V instance = read<V>(state, -1);
